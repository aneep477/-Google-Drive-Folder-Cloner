function doGet(e) { // Paparkan UI Utama jika tiada arahan salin if (!e.parameter.source) { return HtmlService.createTemplateFromFile('index') .evaluate() .setTitle('Drive Cloner Pro') .addMetaTag('viewport', 'width=device-width, initial-scale=1') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } try { // 1. Akses Sheet Aktif const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheets()[0]; // 2. Ekstrak ID dari URL yang dimasukkan const sourceId = extractId(e.parameter.source); const targetId = extractId(e.parameter.target); // 3. Catat Log Permulaan ke Sheet sheet.appendRow([new Date(), "🚀 MEMULAKAN", "Sumber: " + sourceId, "Sasaran: " + targetId]); // 4. Proses Penyalinan const sourceFolder = DriveApp.getFolderById(sourceId); const targetFolder = DriveApp.getFolderById(targetId); copyFolder(sourceFolder, targetFolder); // 5. Catat Log Selesai sheet.appendRow([new Date(), "✅ SELESAI", "Status: Berjaya", "Sistem: Automasi Pro"]); return ContentService.createTextOutput("Success"); } catch (err) { // Catat Ralat jika berlaku const ss = SpreadsheetApp.getActiveSpreadsheet(); ss.getSheets()[0].appendRow([new Date(), "❌ RALAT", err.message]); return ContentService.createTextOutput("Error: " + err.message); } } function extractId(url) { const match = url.match(/[-\w]{25,}/); return match ? match[0] : url; } function copyFolder(source, target) { const files = source.getFiles(); while (files.hasNext()) { files.next().makeCopy(target); } const subFolders = source.getFolders(); while (subFolders.hasNext()) { const sub = subFolders.next(); const newSub = target.createFolder(sub.getName()); copyFolder(sub, newSub); } }
